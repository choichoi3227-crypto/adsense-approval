MySQL/MariaDB 기본 운영 명령어 정리

애플리케이션을 운영하다 보면 직접 DB에 접속해 확인하거나 조작해야 하는 순간이 반드시 옵니다.

접속과 기본 탐색

# 로컬 root 접속
mysql -u root -p

# 데이터베이스 목록
SHOW DATABASES;

# 사용할 DB 선택
USE mydb;

# 테이블 목록
SHOW TABLES;

데이터베이스와 사용자 생성

CREATE DATABASE myapp CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE USER 'appuser'@'localhost' IDENTIFIED BY 'strong_password';

GRANT ALL PRIVILEGES ON myapp.* TO 'appuser'@'localhost';

FLUSH PRIVILEGES;

백업과 복구

# 백업
mysqldump -u root -p myapp > myapp_backup.sql

# 백업 (압축)
mysqldump -u root -p myapp | gzip > myapp_$(date +%Y%m%d).sql.gz

# 복구
mysql -u root -p myapp < myapp_backup.sql

느린 쿼리 로그 활성화

# my.cnf에 추가
slow_query_log = 1
slow_query_log_file = /var/log/mysql/slow.log
long_query_time = 1

1초 이상 걸리는 쿼리를 로그로 남겨 성능 병목을 찾는 데 활용합니다.

프로세스 목록과 잠금 확인

-- 실행 중인 쿼리 목록
SHOW PROCESSLIST;

-- 잠금 상태 확인
SHOW ENGINE INNODB STATUS\G

쿼리가 멈춰 있거나 응답이 없을 때 프로세스 목록을 확인해 문제 쿼리를 찾고 KILL 프로세스ID로 종료할 수 있습니다.
